    打包后的 js 很明显是一个自定义执行函数，代码也不多，代码从上往下可以分析为以下这几个步骤

    1.定义了一个对象，里面的内容就是：模块的路径(key): 函数(value)的形式

    2.定义一个对象, 作为加载模块的缓存

    3.定义一个函数, 当我们加载一个模块时, 都会通过这个函数来加载

    4.调用__webpack_require__执行代码

    1. __webpack_require__：手动实现加载一个模块。对已加载过的模块进行缓存，对未加载过的模块，执行 id 定位到 __webpack_modules__ 中的包裹函数，执行并返回 module.exports，并缓存

    2. 执行__webpack_require__('./src/js/format.js')代码其实就是返回了一个对象,里面含有 dateFormat和priceFormat这两个函数，然后就可以调用成功了

    代码如下：

    // 1.定义了一个对象
// 模块的路径(key): 函数(value)
var __webpack_modules__ = {
  './src/js/format.js': function (module) {
    const dateFormat = (date) => {
      return '2020-12-12'
    }
    const priceFormat = (price) => {
      return '100.00'
    }

    // 将我们要导出的变量, 放入到module对象中的exports对象
    module.exports = {
      dateFormat,
      priceFormat,
    }
  },
}

// 2.定义一个对象, 作为加载模块的缓存
var __webpack_module_cache__ = {}

// 3.定义一个函数, 当我们加载一个模块时, 都会通过这个函数来加载
function __webpack_require__(moduleId) {
  // 判断缓存中是否已经加载过
  if (__webpack_module_cache__[moduleId]) {
    return __webpack_module_cache__[moduleId].exports
  }

  // 给module变量和__webpack_module_cache__[moduleId]赋值了同一个对象
  var module = (__webpack_module_cache__[moduleId] = { exports: {} })

  // 加载执行模块
  __webpack_modules__[moduleId](module, module.exports, __webpack_require__)

  // 导出module.exports {dateFormat: function, priceForamt: function}
  return module.exports
}

// 4.开始执行代码逻辑
!(function () {
  // 加载./src/js/format.js
  const { dateFormat, priceFormat } = __webpack_require__('./src/js/format.js')
  console.log(dateFormat('abc'))
  console.log(priceFormat('abc'))
})()

    和 commonjs 转换的代码的步骤其实都差不多，只是多了一些辅助函数

    1.定义了一个对象，里面的内容就是：模块的路径(key): 函数(value)的形式

    2.定义一个对象, 作为加载模块的缓存

    3.定义一个函数, 当我们加载一个模块时, 都会通过这个函数来加载

    4.调用__webpack_require__执行代码

    1. __webpack_require__：手动实现加载一个模块。对已加载过的模块进行缓存，对胃加载过的模块，执行 id 定位到 __webpack_modules__ 中的包裹函数，执行并返回 module.exports，并缓存

    2. 此时执行__webpack_modules__的代码会执行以下操作:

      1. 调用__webpack_require__.r 将./src/js/math.js 记录是一个 esModule 函数
      2. 调用__webpack_require__.d 给 exports 对象设置了一个代理，然后每一个 key 值都有对应的 getter 函数，当调用的时候就能获取到我们写的函数了
    // 1.定义了一个对象，里面的内容就是：模块的路径(key): 函数(value)的形式
var __webpack_modules__ = {
  './src/js/math.js': (
    __unused_webpack_module,
    __webpack_exports__,
    __webpack_require__
  ) => {
    // 6.将exports标识为es module
    __webpack_require__.r(__webpack_exports__)
    // 7.给exports对象进行代理 从而能调用到方法
    __webpack_require__.d(__webpack_exports__, {
      sum: () => sum,
      mul: () => mul,
    })
    const sum = (num1, num2) => {
      return num1 + num2
    }

    const mul = (num1, num2) => {
      return num1 * num2
    }
  },
}

// 2.定义一个对象, 作为加载模块的缓存
var __webpack_module_cache__ = {}

// 3.定义一个函数, 当我们加载一个模块时, 都会通过这个函数来加载 对已加载过的模块进行缓存，对未加载过的模块，执行 id 定位到 webpack_modules 中的包裹函数，执行并返回 module.exports，并缓存
function __webpack_require__(moduleId) {
  var cachedModule = __webpack_module_cache__[moduleId]
  if (cachedModule !== undefined) {
    return cachedModule.exports
  }
  var module = (__webpack_module_cache__[moduleId] = {
    exports: {},
  })
  // 5.调用__webpack_modules__中所存储的方法
  __webpack_modules__[moduleId](module, module.exports, __webpack_require__)
  // 8. 返回定义好的export对象
  return module.exports
}
// 给exports对象中的属性都添加代理
;(() => {
  __webpack_require__.d = (exports, definition) => {
    for (var key in definition) {
      if (
        __webpack_require__.o(definition, key) &&
        !__webpack_require__.o(exports, key)
      ) {
        Object.defineProperty(exports, key, {
          enumerable: true,
          get: definition[key],
        })
      }
    }
  }
})()

// 这个方法会查找一个对象是否有某个属性，但是不会去查找它的原型链。
;(() => {
  __webpack_require__.o = (obj, prop) =>
    Object.prototype.hasOwnProperty.call(obj, prop)
})()

;(() => {
  __webpack_require__.r = (exports) => {

    其实很简单，就是上面 1 和 2 中 webpack 转换的代码：

    • math.js 是 es module 导出的代码，这个函数最终返回的是 exports 这个代理了的函数所以里面其实是有 mul 和 sum 这两个 key 的 而这两个 key 通过 Object.defineProperty 的 get 函数可以获取到参数，所以 webpack 能结构调用成功
